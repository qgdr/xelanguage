# 特性 Trait

```
struct HasDrop1;
struct HasDrop2;
impl Drop for HasDrop1 {
    fn drop(self: Self@) {
        println("Dropping HasDrop1!");
    }
}
impl Drop for HasDrop2 {
    fn drop(self: Self@) {
        println("Dropping HasDrop2!");
    }
}
struct HasTwoDrops {
    one: HasDrop1,
    two: HasDrop2,
}
impl Drop for HasTwoDrops {
    fn drop(self: Self@) {
        println("Dropping HasTwoDrops!");
    }
}

struct Foo;

impl Drop for Foo {
    fn drop(self: Self@) {
        println("Dropping Foo!")
    }
}

fn main() {
    let _x = HasTwoDrops {
        one;
        two;
    };
    let _foo = Foo;
    println("Running!");
}
```

输出

```
Running!
Dropping Foo!
Dropping HasTwoDrops!
Dropping HasDrop1!
Dropping HasDrop2!
```
